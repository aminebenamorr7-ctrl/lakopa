const express = require('express'); const cors = require('cors'); const morgan = require('morgan'); const path = require('path'); const axios = require('axios'); require('dotenv').config(); const app = express(); const PORT = process.env.PORT || 80; const AI_SERVICE_URL = process.env.AI_SERVICE_URL || 'http://localhost:5000'; // Import routes const itemsRouter = require('./routes/items'); // Middleware app.use(cors()); app.use(morgan('dev')); app.use(express.json({ limit: '50mb' })); app.use(express.urlencoded({ extended: true, limit: '50mb' })); // Serve static files from public folder app.use(express.static(path.join(__dirname, 'public'))); // ===== API Routes ===== app.use('/api/items', itemsRouter); // ===== AI Service Proxy Routes ===== app.get('/api/ai/health', async (req, res) => { try { const response = await axios.get(`${AI_SERVICE_URL}/api/ai/health`, { timeout: 2000 }); res.json(response.data); } catch (error) { res.json({ status: 'healthy', service: 'Inventory AI Service v2.0', type: 'Rule-Based AI (fallback)', timestamp: new Date().toISOString() }); } }); app.post('/api/ai/categorize', async (req, res) => { try { const { item_name, description } = req.body; if (!item_name) return res.status(400).json({ success: false, message: 'Item name is required' }); const response = await axios.post(`${AI_SERVICE_URL}/api/ai/categorize`, { item_name, description: description || '' }); res.json(response.data); } catch (error) { const text = `${req.body.item_name || ''} ${req.body.description || ''}`.toLowerCase(); const keywords = { 'Electronics': ['phone', 'laptop', 'computer', 'charger', 'cable', 'usb'], 'Clothing': ['shirt', 'pants', 'dress', 'shoe', 'jacket', 'cloth'], 'Food': ['food', 'drink', 'snack', 'coffee', 'tea', 'organic'], 'Office Supplies': ['paper', 'pen', 'pencil', 'desk', 'office', 'notebook'] }; let bestCat = 'Other', bestScore = 0; for (const [cat, words] of Object.entries(keywords)) { const score = words.filter(w => text.includes(w)).length; if (score > bestScore) { bestScore = score; bestCat = cat; } } res.json({ success: true, data: { category: bestCat, confidence: Math.min(bestScore / 3, 1.0), all_predictions: [{ category: bestCat, confidence: Math.min(bestScore / 3, 1.0) }] } }); } }); app.post('/api/ai/predict-demand', async (req, res) => { try { const { item_id, item_name, current_stock, sales_history } = req.body; const response = await axios.post(`${AI_SERVICE_URL}/api/ai/predict-demand`, { item_id, item_name, current_stock: current_stock || 0, sales_history: sales_history || [] }); res.json(response.data); } catch (error) { const stock = req.body.current_stock || 0; const avgDaily = stock > 100 ? 5 : stock > 10 ? 2 : 0.5; res.json({ success: true, data: { item_id: req.body.item_id, item_name: req.body.item_name || 'Unknown', current_stock: stock, avg_daily_sales: avgDaily, predicted_demand_7d: Math.round(avgDaily * 7), predicted_demand_30d: Math.round(avgDaily * 30), reorder_point: Math.round(avgDaily * 7 * 1.5), optimal_stock: Math.round(avgDaily * 30 * 1.2), stock_status: stock < avgDaily * 7 * 1.5 ? 'low' : 'optimal', recommendation: stock < avgDaily * 7 * 1.5 ? 'Reorder soon' : 'Stock adequate' } }); } }); app.post('/api/ai/detect-anomalies', async (req, res) => { try { const { items } = req.body; const response = await axios.post(`${AI_SERVICE_URL}/api/ai/detect-anomalies`, { items }); res.json(response.data); } catch (error) { const items = req.body.items || []; const anomalies = []; for (const item of items) { const issues = []; if (item.quantity < 0) issues.push("Negative stock quantity"); if (item.price < 0) issues.push("Negative price"); if (item.price > 10000) issues.push("Unusually high price"); if (item.quantity > 10000) issues.push("Unusually high stock"); if (item.quantity === 0) issues.push("Out of stock"); if (issues.length > 0) anomalies.push({ item_id: item.id, item_name: item.item_name || 'Unknown', issues, severity: issues.length > 1 ? 'high' : 'medium' }); } res.json({ success: true, data: anomalies, count: anomalies.length }); } }); app.post('/api/ai/optimize-price', async (req, res) => { try { const response = await axios.post(`${AI_SERVICE_URL}/api/ai/optimize-price`, req.body); res.json(response.data); } catch (error) { const { item_name, current_price, quantity } = req.body; const suggested = quantity > 100 ? current_price * 0.95 : quantity < 10 ? current_price * 1.1 : current_price; res.json({ success: true, data: { item_name: item_name || 'Unknown', current_price, suggested_price: Math.round(suggested * 100) / 100, potential_profit_change: Math.round((suggested - current_price) * (quantity || 0) * 100) / 100, price_change_percent: Math.round(((suggested - current_price) / current_price) * 10000) / 100, reasoning: quantity > 100 ? 'High stock - slight discount' : quantity < 10 ? 'Low stock - premium price' : 'Price is optimal' } }); } }); app.post('/api/ai/inventory-health', async (req, res) => { try { const response = await axios.post(`${AI_SERVICE_URL}/api/ai/inventory-health`, req.body); res.json(response.data); } catch (error) { const items = req.body.items || []; const total = items.length; if (total === 0) return res.json({ success: true, data: { status: 'empty', score: 0, metrics: { total_items: 0, out_of_stock: 0, low_stock: 0, missing_category: 0, missing_price: 0 }, recommendations: [{ priority: 'high', action: 'Add items', impact: 'No items' }] } }); const outOfStock = items.filter(i => i.quantity === 0).length; const lowStock = items.filter(i => i.quantity > 0 && i.quantity < 10).length; const noCategory = items.filter(i => !i.category).length; const noPrice = items.filter(i => !i.price).length; let score = 100 - (outOfStock / total) * 30 - (lowStock / total) * 20 - (noCategory / total) * 15 - (noPrice / total) * 15; score = Math.max(0, Math.min(100, Math.round(score))); const status = score >= 80 ? 'excellent' : score >= 60 ? 'good' : score >= 40 ? 'fair' : 'poor'; res.json({ success: true, data: { status, score, metrics: { total_items: total, out_of_stock: outOfStock, low_stock: lowStock, missing_category: noCategory, missing_price: noPrice }, recommendations: [] } }); } }); // ===== Frontend - Catch all routes ===== app.get('*', (req, res) => { res.sendFile(path.join(__dirname, 'public', 'index.html')); }); // ===== Error Handling ===== app.use((err, req, res, next) => { console.error('Error:', err.stack); res.status(500).json({ success: false, message: 'Server error', error: err.message }); }); // ===== Start Server ===== app.listen(PORT, () => { console.log('='.repeat(50)); console.log(`🚀 Server running on http://localhost:${PORT}`); console.log(`📁 Static files: ${path.join(__dirname, 'public')}`); console.log(`📦 API: http://localhost:${PORT}/api/items`); console.log(`🤖 AI: http://localhost:${PORT}/api/ai`); console.log('='.repeat(50)); });
